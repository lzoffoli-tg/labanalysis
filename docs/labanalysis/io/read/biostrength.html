115        """return the calculated lever weight"""
116        return (
117            self.load_motor_nm

121            / self.spring_correction
122            + self.lever_weight_kgf

130        num = self._position_motor_rad[:-2] - self._position_motor_rad[2:]
131        den = self._time_s[:-2] - self._time_s[2:]
132        return (num / den).astype(float)

228    def __init__(
229        self,
230        time_s: NDArray[np.floating],
231        motor_position_rad: NDArray[np.floating],
232        load_kgf: NDArray[np.floating],
233    ):
234        # check the entries
235        try:
236            self._time_s = np.array([time_s]).astype(float).flatten()
237        except Exception as exc:
238            raise ValueError(
239                "time must be castable to a numpy array of floats"
240            ) from exc
241        try:
242            self._position_motor_rad = (
243                np.array([motor_position_rad]).astype(float).flatten()
244            )
245        except Exception as exc:
246            raise ValueError(
247                "motor_position_rad must be castable to a numpy array of floats"
248            ) from exc
249        try:
250            self._load_motor_nm = (
251                (load_kgf - self._lever_weight_kgf)
252                * G
253                * self._pulley_radius_m
254                * self._spring_correction
255                / self._camme_ratio
256            )
257
258        except Exception as exc:
259            raise ValueError(
260                "motor_load_nm must be castable to a numpy array of floats"
261            ) from exc
262
263        # check the length of each element
264        if (
265            not len(self.time_s)
266            == len(self.position_motor_rad)
267            == len(self.load_motor_nm)
268        ):
269            msg = "time_s, motor_position_rad and motor_load_nm must all have "
270            msg += "the same number of samples."
271            raise ValueError(msg)
