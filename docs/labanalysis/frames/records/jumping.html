 98    def _get_symmetry(self, left: np.ndarray, right: np.ndarray):
 99        left_val = np.mean(left)
100        right_val = np.mean(right)
101        vals = left_val + right_val
102        return {
103            "left_%": left_val / vals * 100,
104            "right_%": right_val / vals * 100,
105        }

107    def _get_muscle_symmetry(self):
108        """
109        Returns coordination and balance metrics from EMG signals.
110
111        Returns
112        -------
113        pd.DataFrame
114            DataFrame with coordination and balance metrics, or empty if not available.
115        """
116
117        # check if a bilateral jump was performed
118        # (otherwise it makes no sense to test balance)
119        if self.side != "bilateral":
120            return pd.DataFrame()
121
122        # get the muscle activations
123        # (if there are no emg data return and empty dataframe)
124        emgs = self.contact_phase.emgsignals
125        if emgs.shape[1] == 0:
126            return pd.DataFrame()
127
128        # check the presence of left and right muscles
129        muscles = {}
130        for emg in emgs.values():
131            if isinstance(emg, EMGSignal):
132                name = emg.muscle_name
133                side = emg.side
134                if side not in ["left", "right"]:
135                    continue
136                if name not in list(muscles.keys()):
137                    muscles[name] = {}
138
139            # get the area under the curve of the muscle activation
140            muscles[name][side] = emg.copy().to_numpy().flatten()
141
142        # remove those muscles not having both sides
143        muscles = {i: v for i, v in muscles.items() if len(v) == 2}
144
145        # calculate coordination and imbalance between left and right side
146        out = {}
147        for muscle, sides in muscles.items():
148            params = self._get_symmetry(**sides)
149            out.update(**{f"{muscle}_{i}": v for i, v in params.items()})
150
151        return pd.DataFrame(pd.Series(out)).T

153    def _get_force_symmetry(self):
154        """
155        Returns coordination and balance metrics from force signals.
156
157        Returns
158        -------
159        pd.DataFrame
160            DataFrame with force coordination and balance metrics, or empty if not available.
161        """
162
163        # get the forces from each foot and hand
164        left_foot = self.get("left_foot_ground_reaction_force")
165        right_foot = self.get("right_foot_ground_reaction_force")
166        if left_foot is None or right_foot is None:
167            return pd.DataFrame()
168        left_foot = left_foot["force"].copy()[self.vertical_axis].to_numpy().flatten()
169        right_foot = right_foot["force"].copy()[self.vertical_axis].to_numpy().flatten()
170
171        # get the pairs to be tested
172        pairs = {"lower_limbs": {"left_foot": left_foot, "right_foot": right_foot}}
173
174        # calculate balance and coordination
175        out = []
176        for region, pair in pairs.items():
177            left, right = list(pair.values())
178            fit = self._get_symmetry(left, right)
179            line = {f"force_{i}": v for i, v in fit.items()}
180            line = pd.DataFrame(pd.Series(line)).T
181            line.insert(0, "region", region)
182            out += [line]
183
184        return pd.concat(out, ignore_index=True)

218    @property
219    def contact_phase(self):
220        """
221        Returns the concentric phase of the jump.
222
223        Returns
224        -------
225        TimeseriesRecord
226            Data for the concentric phase.
227
228        Procedure
229        ---------
230            1. get the longest countinuous batch with positive acceleration
231            of S2 occurring before con_end.
232            2. define 'con_start' as the last local minima in the vertical grf
233            occurring before the beginning of the batch defined in 2.
234            3. define 'con_end' as the end of the concentric phase as the time
235            instant immediately before the flight phase. Please look at the
236            concentric_phase documentation to have a detailed view about how
237            it is detected.
238        """
239        out = WholeBody()
240
241        # get the longest batch with grf lower than 30N
242        vgrf = self.resultant_force
243        if vgrf is None:
244            return out
245        grfy = vgrf.copy().force[self.vertical_axis].to_numpy().flatten()
246        grft = self.index
247        batches = continuous_batches(
248            (grfy > MINIMUM_CONTACT_FORCE_N) & (grft < self.flight_phase.index[0])
249        )
250        if len(batches) == 0:
251            return out
252        batch = batches[-1]
253
254        sliced = self.copy()[grft[batch[0]] : grft[batch[-1]]]
255        if sliced is not None:
256            if isinstance(sliced, WholeBody):
257                for key, value in sliced.items():
258                    out[key] = value
259        return out

261    @property
262    def flight_phase(self):
263        """
264        Returns the flight phase of the jump.
265
266        Returns
267        -------
268        TimeseriesRecord
269            Data for the flight phase.
270
271        Procedure
272        ---------
273            1. get the longest batch with grf lower than 30N.
274            2. define 'flight_start' as the first local minima occurring after
275            the start of the detected batch.
276            3. define 'flight_end' as the last local minima occurring before the
277            end of the detected batch.
278        """
279
280        # get the longest batch with grf lower than 30N
281        vgrf = self.resultant_force.copy()
282        if vgrf is None:
283            return TimeseriesRecord()
284        grfy = vgrf.force.copy()[self.vertical_axis].to_numpy().flatten()
285        grft = self.index
286        batches = continuous_batches(grfy <= MINIMUM_CONTACT_FORCE_N)
287        msg = "No flight phase found."
288        if len(batches) == 0:
289            raise RuntimeError(msg)
290        batch = batches[np.argmax([len(i) for i in batches])]
291
292        # check the length of the batch is at minimum 2 samples
293        if len(batch) < 2:
294            raise RuntimeError(msg)
295
296        # # get the time samples corresponding to the start and end of each
297        # batch
298        time_start = float(np.round(grft[batch[0]], 3))
299        time_stop = float(np.round(grft[batch[-1]], 3))
300
301        # return a slice of the available data
302        sliced = self.copy()[time_start:time_stop]
303        if sliced is None:
304            raise RuntimeError(msg)
305        out = WholeBody()
306        if isinstance(sliced, WholeBody):
307            for key, value in sliced.items():
308                out[key] = value
309        return out

337    @property
338    def takeoff_velocity_ms(self):
339        """
340        Returns the takeoff velocity at the end of the concentric phase.
341
342        Returns
343        -------
344        float
345            Takeoff velocity in m/s.
346        """
347
348        # get the ground reaction force during the concentric phase
349        con = self.contact_phase.resultant_force
350        if con is None:
351            return np.nan
352        grf = con.copy().force[self.vertical_axis].to_numpy().flatten()
353        time = con.index
354
355        # get the output velocity
356        net_grf = grf - self.bodymass_kg * G
357        return float(np.trapezoid(net_grf, time) / self.bodymass_kg)

359    @property
360    def elevation_cm(self):
361        """
362        Returns the jump elevation in centimeters, calculated from flight time.
363
364        Returns
365        -------
366        float
367            Jump elevation in cm.
368        """
369
370        # from flight time
371        flight_time = self.flight_phase.index
372        flight_time = flight_time[-1] - flight_time[0]
373        elevation_from_time = (flight_time**2) * G / 8 * 100
374
375        # from force impulse
376        elevation_from_velocity = (self.takeoff_velocity_ms**2) / (2 * G) * 100
377
378        # return the lower of the two
379        return min(elevation_from_time, elevation_from_velocity)

509        }
510        if all([i is None for i in mandatory_labels.values()]):

512                f"at least one of left_ground_reaction_force or"

516        mandatory = {}
517        for key, lbl in mandatory_labels.items():

521                    raise ValueError(f"{lbl} not found in the provided file.")
522        signals = {i: v for i, v in record.items() if i not in list(mandatory.keys())}
523        return cls(
524            bodymass_kg=bodymass_kg,
525            **signals,  # type: ignore
526            **mandatory,  # type: ignore
527        )

566    @property
567    def jumps(self):
568        jumps: list[SingleJump] = []
569        vgrf = self.resultant_force
570        if vgrf is None:
571            raise ValueError("No resultant force was found")
572        time = np.array(self.index)
573        vgrf = vgrf.force.copy()[self.vertical_axis].to_numpy().flatten()
574
575        # get the batches with grf lower than 30N (i.e flight phases)
576        batches = continuous_batches(vgrf <= float(MINIMUM_CONTACT_FORCE_N))
577
578        # remove those batches resulting in too short flight phases
579        # a jump is assumed valid if the elevation is higher than 5 cm
580        # (i.e. ~0.2s flight time)
581        fsamp = 1 / np.mean(np.diff(time))
582        min_samples = int(round(MINIMUM_FLIGHT_TIME_S * fsamp))
583        batches = [i for i in batches if len(i) >= min_samples]
584
585        # ensure that the first jump does not start with a flight
586        if batches[0][0] == 0:
587            batches = batches[1:]
588
589        # ensure that the last jump does not end in flight
590        if batches[-1][-1] == len(vgrf) - 1:
591            batches = batches[:-1]
592
593        # separate each jump
594        flights_idx = np.where(vgrf < MINIMUM_CONTACT_FORCE_N)[0]
595        for batch in batches:
596            start_idx = flights_idx[flights_idx < batch[0]]
597            start_idx = int(0 if len(start_idx) == 0 else (start_idx[-1] + 1))
598            start = float(time[start_idx])
599            stop = float(time[batch[-1] + 2])
600            sliced = self.copy()[start:stop]
601            if sliced is not None:
602                if isinstance(sliced, TimeseriesRecord):
603                    new_jump = SingleJump(
604                        bodymass_kg=self.bodymass_kg,
605                        **{i: v for i, v in sliced.items()},  # type: ignore
606                    )
607                    jumps += [new_jump]
608
609        return jumps

137                if name not in list(muscles.keys()):
138                    muscles[name] = {}
139
140            # get the area under the curve of the muscle activation
141            muscles[name][side] = emg.copy().to_numpy().flatten()
142
143        # remove those muscles not having both sides
144        muscles = {i: v for i, v in muscles.items() if len(v) == 2}
145
146        # calculate coordination and imbalance between left and right side
147        out = {}
148        for muscle, sides in muscles.items():
149            params = self._get_symmetry(**sides)
150            out.update(**{f"{muscle}_{i}": v for i, v in params.items()})
151
152        return pd.DataFrame(pd.Series(out)).T

220    def contact_phase(self):
221        """
222        Returns the concentric phase of the jump.
223
224        Returns
225        -------
226        TimeseriesRecord
227            Data for the concentric phase.
228
229        Procedure
230        ---------
231            1. get the longest countinuous batch with positive acceleration
232            of S2 occurring before con_end.
233            2. define 'con_start' as the last local minima in the vertical grf
234            occurring before the beginning of the batch defined in 2.
235            3. define 'con_end' as the end of the concentric phase as the time
236            instant immediately before the flight phase. Please look at the
237            concentric_phase documentation to have a detailed view about how
238            it is detected.
239        """
240        out = WholeBody()
241
242        # get the longest batch with grf lower than 30N
243        vgrf = self.resultant_force
244        if vgrf is None:
245            return out
246        grfy = vgrf.copy().force[self.vertical_axis].to_numpy().flatten()
247        grft = self.index
248        batches = continuous_batches(
249            (grfy > MINIMUM_CONTACT_FORCE_N) & (grft < self.flight_phase.index[0])
250        )
251        if len(batches) == 0:
252            return out
253        batch = batches[-1]
254
255        sliced = self.copy()[grft[batch[0]] : grft[batch[-1]]]
256        if sliced is not None:
257            if isinstance(sliced, WholeBody):
258                for key, value in sliced.items():
259                    out[key] = value
260        return out

262    @property
263    def flight_phase(self):
264        """
265        Returns the flight phase of the jump.
266
267        Returns
268        -------
269        TimeseriesRecord
270            Data for the flight phase.
271
272        Procedure
273        ---------
274            1. get the longest batch with grf lower than 30N.
275            2. define 'flight_start' as the first local minima occurring after
276            the start of the detected batch.
277            3. define 'flight_end' as the last local minima occurring before the
278            end of the detected batch.
279        """
280
281        # get the longest batch with grf lower than 30N
282        vgrf = self.resultant_force.copy()
283        if vgrf is None:
284            return TimeseriesRecord()
285        grfy = vgrf.force.copy()[self.vertical_axis].to_numpy().flatten()
286        grft = self.index
287        batches = continuous_batches(grfy <= MINIMUM_CONTACT_FORCE_N)
288        msg = "No flight phase found."
289        if len(batches) == 0:
290            raise RuntimeError(msg)
291        batch = batches[np.argmax([len(i) for i in batches])]
292
293        # check the length of the batch is at minimum 2 samples
294        if len(batch) < 2:
295            raise RuntimeError(msg)
296
297        # # get the time samples corresponding to the start and end of each
298        # batch
299        time_start = float(np.round(grft[batch[0]], 3))
300        time_stop = float(np.round(grft[batch[-1]], 3))
301
302        # return a slice of the available data
303        sliced = self.copy()[time_start:time_stop]
304        if sliced is None:
305            raise RuntimeError(msg)
306        out = WholeBody()
307        if isinstance(sliced, WholeBody):
308            for key, value in sliced.items():
309                out[key] = value
310        return out

340        """
341        Returns the takeoff velocity at the end of the concentric phase.
342
343        Returns
344        -------
345        float
346            Takeoff velocity in m/s.
347        """
348
349        # get the ground reaction force during the concentric phase
350        con = self.contact_phase.resultant_force
351        if con is None:
352            return np.nan
353        grf = con.copy().force[self.vertical_axis].to_numpy().flatten()
354        time = con.index
355
356        # get the output velocity
357        net_grf = grf - self.bodymass_kg * G
358        return float(np.trapezoid(net_grf, time) / self.bodymass_kg)

362        """
363        Returns the jump elevation in centimeters, calculated from flight time.
364
365        Returns
366        -------
367        float
368            Jump elevation in cm.
369        """
370
371        # from flight time
372        flight_time = self.flight_phase.index
373        flight_time = flight_time[-1] - flight_time[0]
374        elevation_from_time = (flight_time**2) * G / 8 * 100
375
376        # from force impulse
377        elevation_from_velocity = (self.takeoff_velocity_ms**2) / (2 * G) * 100
378
379        # return the lower of the two
