242    def _get_grf_positive_crossing_times(self):
243        """
244        Find the positive crossings over the mean force.
245
246        Returns
247        -------
248        np.ndarray
249            Array of positive crossing times.
250
251        Raises
252        ------
253        ValueError
254            If no ground reaction force data is available.
255        """
256        # get the ground reaction force
257        res = self.resultant_force
258        if res is None:
259            raise ValueError("no ground reaction force data available.")
260        time = res.index
261        vres = res.force.copy()[self.vertical_axis].to_numpy().flatten()
262        vres -= np.nanmean(vres)
263        vres /= np.max(vres)
264
265        # get the zero-crossing points
266        zeros, signs = crossings(vres, 0)
267        return time[zeros[signs > 0]].astype(float)

288    def _footstrike_kinematics(self):
289        """
290        Find the footstrike time using the kinematics algorithm.
291
292        Returns
293        -------
294        float
295            The footstrike time in seconds.
296
297        Raises
298        ------
299        ValueError
300            If no footstrike has been found.
301        """
302
303        # get the relevant vertical coordinates
304        vcoords = {}
305        time = self.index
306        for marker in ["heel", "metatarsal_head"]:
307            lbl = f"{self.side.lower()}_{marker}"
308            dfr = getattr(self, lbl)
309            if dfr is not None:
310                vcoords[lbl] = dfr.copy()[self.vertical_axis].to_numpy().flatten()
311
312        # filter the signals and extract the first contact time
313        fs_time = []
314        for val in vcoords.values():
315            val = val / np.max(val)
316            fsi = np.where(val < self.height_threshold)[0]
317            if len(fsi) > 0:
318                fs_time += [time[fsi[0]]]
319
320        # return
321        if len(fs_time) == 0:
322            raise ValueError("not footstrike has been found")
323        return float(np.min(fs_time))

365    def _midstance_kinetics(self):
366        """
367        Find the midstance time using the kinetics algorithm.
368
369        Returns
370        -------
371        float
372            The midstance time in seconds.
373
374        Raises
375        ------
376        ValueError
377            If resultant_force not found or no valid mid-stance was found.
378        """
379
380        # get the anterior-posterior resultant force
381        res = self.resultant_force
382        if res is None:
383            raise ValueError("resultant_force not found")
384        time = res.index
385        res_ap = res.copy()[self.anteroposterior_axis].to_numpy().flatten()  # type: ignore
386        res_ap -= np.nanmean(res_ap)
387
388        # get the dominant frequency
389        fsamp = float(1 / np.mean(np.diff(time)))
390        frq, pwr = psd(res_ap, fsamp)
391        ffrq = frq[np.argmax(pwr)]
392
393        # find the local minima
394        min_samp = int(fsamp / ffrq / 2)
395        mns = find_peaks(-res_ap, 0, min_samp)
396        if len(mns) != 2:
397            raise ValueError("no valid mid-stance was found.")
398        pk = np.argmax(res_ap[mns[0] : mns[1]]) + mns[0]
399
400        # get the range and obtain the toe-off
401        # as the last value occurring before the peaks within the
402        # 1 - height_threshold of that range
403        thresh = (1 - self.height_threshold) * res_ap[pk]
404        loc = np.where(res_ap[pk:] < thresh)[0] + pk
405        if len(loc) > 0:
406            return float(time[loc[0]])
407        raise ValueError("no valid mid-stance was found.")

409    def _midstance_kinematics(self):
410        """
411        Find the midstance time using the kinematics algorithm.
412
413        Returns
414        -------
415        float
416            The midstance time in seconds.
417        """
418
419        # get the minimum height across all foot markers
420        vcoord = []
421        time = self.index
422        for lbl in ["heel", "metatarsal_head", "toe"]:
423            name = f"{self.side.lower()}_{lbl}"
424            val = getattr(self, name)
425            if val is not None:
426                val = val.copy()[self.vertical_axis].to_numpy().flatten()
427                val = val / np.max(val)
428                vcoord += [val]
429        vcoord = np.vstack(np.atleast_2d(*vcoord)).mean(axis=0)
430        idx = np.argmin(vcoord)
431        return float(time[idx])

633    def _find_cycles_kinematics(self):
634        """
635        Find the gait cycles using the kinematics algorithm.
636
637        Returns
638        -------
639        None
640
641        Raises
642        ------
643        ValueError
644            If any required marker is missing or no toe-offs have been found.
645        """
646
647        # get toe-off times
648        time = self.index
649        fsamp = float(1 / np.mean(np.diff(time)))
650        cycles: list[WalkingStride] = []
651        for lbl in ["left_toe", "right_toe"]:
652
653            # get the vertical coordinates of the toe markers
654            arr = self[lbl].copy()[self.vertical_axis].to_numpy().flatten()  # type: ignore
655
656            # filter and rescale
657            ftoe = arr / np.max(arr)
658
659            # get the minimum reasonable contact time for each step
660            frq, pwr = psd(ftoe, fsamp)
661            ffrq = frq[np.argmax(pwr)]
662            dsamples = int(round(fsamp / ffrq / 2))
663
664            # get the peaks at each cycle
665            pks = find_peaks(ftoe, 0.5, dsamples)
666
667            # for each peak obtain the location of the last sample at the
668            # required height threshold
669            tos = []
670            side = lbl.split("_")[0]
671            for pk in pks:
672                idx = np.where(ftoe[:pk] <= self.height_threshold)[0]
673                if len(idx) > 0:
674                    line = pd.Series({"time": time[idx[-1]], "side": side})
675                    tos += [pd.DataFrame(line).T]
676
677            # wrap the events
678            if len(tos) == 0:
679                raise ValueError("no toe-offs have been found.")
680            tos = pd.concat(tos, ignore_index=True)
681            tos = tos.drop_duplicates()
682            tos = tos.sort_values("time")
683            tos = tos.reset_index(drop=True)

354        fs_time = []
355        for val in vcoords.values():
356            val = val / np.max(val)
357            fsi = np.where(val >= self.height_threshold)[0]
358            if len(fsi) > 0 and fsi[-1] + 1 < len(time):
359                fs_time += [time[fsi[-1] + 1]]
360
361        # return
362        if len(fs_time) == 0:
363            raise ValueError("not opposite footstrike has been found")
364        return float(np.min(fs_time))

366    def _midstance_kinetics(self):
367        """
368        Find the midstance time using the kinetics algorithm.
369
370        Returns
371        -------
372        float
373            The midstance time in seconds.
374
375        Raises
376        ------
377        ValueError
378            If resultant_force not found or no valid mid-stance was found.
379        """
380
381        # get the anterior-posterior resultant force
382        res = self.resultant_force
383        if res is None:
384            raise ValueError("resultant_force not found")
385        time = res.index
386        res_ap = res.copy()[self.anteroposterior_axis].to_numpy().flatten()  # type: ignore
387        res_ap -= np.nanmean(res_ap)
388
389        # get the dominant frequency
390        fsamp = float(1 / np.mean(np.diff(time)))
391        frq, pwr = psd(res_ap, fsamp)
392        ffrq = frq[np.argmax(pwr)]
393
394        # find the local minima
395        min_samp = int(fsamp / ffrq / 2)
396        mns = find_peaks(-res_ap, 0, min_samp)
397        if len(mns) != 2:
398            raise ValueError("no valid mid-stance was found.")
399        pk = np.argmax(res_ap[mns[0] : mns[1]]) + mns[0]
400
401        # get the range and obtain the toe-off
402        # as the last value occurring before the peaks within the
403        # 1 - height_threshold of that range
404        thresh = (1 - self.height_threshold) * res_ap[pk]
405        loc = np.where(res_ap[pk:] < thresh)[0] + pk
406        if len(loc) > 0:
407            return float(time[loc[0]])
408        raise ValueError("no valid mid-stance was found.")

410    def _midstance_kinematics(self):
411        """
412        Find the midstance time using the kinematics algorithm.
413
414        Returns
415        -------
416        float
417            The midstance time in seconds.
418        """
419
420        # get the minimum height across all foot markers
421        vcoord = []
422        time = self.index
423        for lbl in ["heel", "metatarsal_head", "toe"]:
424            name = f"{self.side.lower()}_{lbl}"
425            val = getattr(self, name)
426            if val is not None:
427                val = val.copy()[self.vertical_axis].to_numpy().flatten()
428                val = val / np.max(val)
429                vcoord += [val]
430        vcoord = np.vstack(np.atleast_2d(*vcoord)).mean(axis=0)
431        idx = np.argmin(vcoord)
432        return float(time[idx])
