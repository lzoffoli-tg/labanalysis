 238    def set_algorithm(self, algorithm: Literal["kinematics", "kinetics"]):
 239        """
 240        Set the gait cycle detection algorithm.
 241
 242        Parameters
 243        ----------
 244        algorithm : {'kinematics', 'kinetics'}
 245            Algorithm label.
 246        """
 247        algorithms = ["kinematics", "kinetics"]
 248        if not isinstance(algorithm, str) or algorithm not in algorithms:
 249            msg = "'algorithm' must be any between 'kinematics' or 'kinetics'."
 250            raise ValueError(msg)
 251        algo = algorithm
 252        if (
 253            algo == "kinetics"
 254            and self.resultant_force is None
 255            and all(
 256                [
 257                    self.left_heel is not None,
 258                    self.left_toe is not None,
 259                    self.right_heel is not None,
 260                    self.right_toe is not None,
 261                ]
 262            )
 263        ):
 264            msg = f"'forceplatforms data' not found. The 'algorithm' option"
 265            msg += " has been set to 'kinematics'."
 266            warnings.warn(msg)
 267            algo = "kinematics"
 268        elif (
 269            algo == "kinematics"
 270            and self.resultant_force is not None
 271            and not all(
 272                [
 273                    self.left_heel is not None,
 274                    self.left_toe is not None,
 275                    self.right_heel is not None,
 276                    self.right_toe is not None,
 277                ]
 278            )
 279        ):
 280            msg = f"Not all left_heel, right_heel, left_toe and right_toe"
 281            msg += " markers have been found to run the 'kinematics' algorithm."
 282            msg += " The 'kinetics' algorithm has therefore been selected."
 283            warnings.warn(msg)
 284            algo = "kinetics"
 285        elif self.resultant_force is None and any(
 286            [
 287                self.left_heel is None,
 288                self.left_toe is None,
 289                self.right_heel is None,
 290                self.right_toe is None,
 291            ]
 292        ):
 293            msg = "Neither ground reaction force nor left_heel, right_heel, "
 294            msg += "left_toe and right_toe markers have been found."
 295            msg += " Therefore none of the available algorithms can be used."
 296            raise ValueError(msg)
 297
 298        self._algorithm = algo

 561    @property

 563        """
 564        Return all the time events defining the cycle.
 565

 567        -------

 569        """
 570        evts: dict[str, float] = {}
 571        for lbl in dir(self):
 572            if lbl.endswith("_s") and not lbl.startswith("_"):
 573                name = lbl.rsplit("_", 1)[0].strip().split(" ")[0].lower()
 574                time = getattr(self, lbl)
 575                perc = time
 576                if lbl in self._absolute_time_events:
 577                    perc -= self.init_s
 578                perc = perc / self.cycle_time_s * 100
 579                evts[f"{name.lower().replace("_time", "")}_s"] = float(time)
 580                evts[f"{name.lower().replace("_time", "")}_%"] = float(perc)
 581        return evts

 583    @property
 584    def lateral_displacement(self):
 585        try:
 586            obj = self.pelvis_center
 587        except Exception as exc:
 588            obj = self.resultant_force
 589            if obj is None:
 590                return np.nan
 591            obj = obj.origin.copy()
 592        arr = obj[self.lateral_axis].to_numpy()
 593        return float(np.max(arr) - np.min(arr))
