120    def _footstrike_kinetics(self):
121        """
122        Find the footstrike time using the kinetics algorithm.
123
124        Returns
125        -------
126        float
127            The footstrike time in seconds.
128
129        Raises
130        ------
131        ValueError
132            If no ground reaction force data is available or no footstrike is found.
133        """
134
135        # get the contact phase samples
136        grf = self.resultant_force
137        if grf is None:
138            raise ValueError("no ground reaction force data available.")
139        vgrf = grf.force.copy()[self.vertical_axis].to_numpy().flatten()
140        time = grf.index
141        grfn = vgrf / np.max(vgrf)
142        mask = np.where(grfn[: np.argmax(grfn)] < self.height_threshold)[0]
143
144        # extract the first contact time
145        if len(mask) == 0:
146            raise ValueError("no footstrike has been found.")
147
148        return float(time[mask[-1]])

150    def _footstrike_kinematics(self):
151        """
152        Find the footstrike time using the kinematics algorithm.
153
154        Returns
155        -------
156        float
157            The footstrike time in seconds.
158
159        Raises
160        ------
161        ValueError
162            If no footstrike has been found.
163        """
164
165        # get the relevant vertical coordinates
166        vcoords = {}
167        contact_foot = self.side.lower()
168        for marker in ["heel", "metatarsal_head"]:
169            lbl = f"{contact_foot}_{marker}"
170            val = self[f"{contact_foot}_{marker}"]
171            if val is None:
172                continue
173            vcoords[lbl] = val.copy()[self.vertical_axis].to_numpy().flatten()  # type: ignore
174
175        # filter the signals and extract the first contact time
176        time = self.index
177        fs_time = []
178        for val in vcoords.values():
179            val = val / np.max(val)
180            fsi = np.where(val < self.height_threshold)[0]
181            if len(fsi) == 0 or fsi[0] == 0:
182                raise ValueError("not footstrike has been found.")
183            fs_time += [time[fsi[0]]]
184
185        # get output time
186        if len(fs_time) > 0:
187            return float(np.min(fs_time))
188        raise ValueError("no footstrike has been found.")

220        """

235
236        # return the time corresponding to the minimum value
237        return float(time[np.argmin(ref)])

372    def _find_cycles_kinematics(self):
373        """
374        Find the gait cycles using the kinematics algorithm.
375
376        Returns
377        -------
378        None
379
380        Raises
381        ------
382        ValueError
383            If any required marker is missing or no toe-offs are found.
384        Warns
385        -----
386        UserWarning
387            If left-right steps alternation is not guaranteed.
388        """
389
390        # get toe-off times
391        tos = []
392        time = self.index
393        fsamp = float(1 / np.mean(np.diff(time)))
394        for lbl in ["left_toe", "right_toe"]:
395
396            # get the vertical coordinates of the toe markers
397            obj = self.get(lbl)
398            if obj is None:
399                raise ValueError(f"{lbl} is missing.")
400            arr = obj.copy()[self.vertical_axis].to_numpy().flatten()
401
402            # filter and rescale
403            arr = arr / np.max(arr)
404
405            # get the minimum reasonable contact time for each step
406            dsamples = int(round(fsamp / 2))
407
408            # get the peaks at each cycle
409            pks = find_peaks(arr, 0.5, dsamples)
410
411            # for each peak obtain the location of the last sample at the
412            # required height threshold
413            side = lbl.split("_")[0]
414            for pk in pks:
415                idx = np.where(arr[:pk] <= self.height_threshold)[0]
416                if len(idx) > 0:
417                    line = pd.Series({"Time": time[idx[-1]], "Side": side})
418                    tos += [pd.DataFrame(line).T]
419
420        # wrap the events
421        if len(tos) == 0:
422            raise ValueError("no toe-offs have been found.")
423        tos = pd.concat(tos, ignore_index=True)
424        tos = tos.drop_duplicates()
425        tos = tos.sort_values("Time")
426        tos = tos.reset_index(drop=True)
427
428        # check the alternation of the steps
429        sides = tos.Side.values
430        if not all(s0 != s1 for s0, s1 in zip(sides[:-1], sides[1:])):
431            warnings.warn("Left-Right steps alternation not guaranteed.")
432
433        return [
434            self._get_cycle(
435                float(tos.Time.values[i0]),
436                float(tos.Time.values[i1]),
437                tos.Side.values[i1],  # type: ignore
438            )
439            for i0, i1 in zip(tos.index[:-1], tos.index[1:])
440        ]

442    def _find_cycles_kinetics(self):
443        """
444        Find the gait cycles using the kinetics algorithm.
445
446        Returns
447        -------
448        None
449
450        Raises
451        ------
452        ValueError
453            If no ground reaction force data is available or no flight phases are found.
454        """
455
456        grf = self.resultant_force
457        if grf is None:
458            raise ValueError("no ground reaction force data available.")
459
460        # get the grf and the latero-lateral COP
461        time = grf.index
462        cop_ml = grf["origin"].copy()[self.lateral_axis].to_numpy().flatten()  # type: ignore
463        vgrf = grf["force"].copy()[self.vertical_axis].to_numpy().flatten()  # type: ignore
464
465        # check if there are flying phases
466        flights = vgrf <= self.ground_reaction_force_threshold
467        if not any(flights):
468            raise ValueError("No flight phases have been found on data.")
469
470        # get the minimum reasonable contact time for each step
471        fsamp = float(1 / np.mean(np.diff(time)))
472        dsamples = int(round(fsamp / 4))
473
474        # get the peaks in the normalized grf, then return toe-offs and foot
475        # strikes
476        grfn = vgrf / np.max(vgrf)
477        toi = []
478        fsi = []
479        pks = find_peaks(grfn, 0.5, dsamples)
480        for pk in pks:
481            to = np.where(grfn[pk:] < self.height_threshold)[0]
482            fs = np.where(grfn[:pk] < self.height_threshold)[0]
483            if len(fs) > 0 and len(to) > 0:
484                toi += [to[0] + pk]
485                if len(toi) > 1:
486                    fsi += [fs[-1]]
487        toi = np.unique(toi)
488        fsi = np.unique(fsi)
489
490        # get the mean latero-lateral position of each contact
491        contacts = [np.arange(i, j + 1) for i, j in zip(fsi, toi[1:])]
492        pos = [np.nanmean(cop_ml[i]) for i in contacts]
493
494        # get the mean value of alternated contacts and set the step sides
495        # accordingly
496        evens = np.mean(pos[0:-1:2])
497        odds = np.mean(pos[1:-1:2])
498        sides = []
499        for i in np.arange(len(pos)):
500            if evens < odds:
501                sides += ["left" if i % 2 == 0 else "right"]
502            else:
503                sides += ["left" if i % 2 != 0 else "right"]
504
505        return [
506            self._get_cycle(float(time[to]), float(time[ed]), sd)
507            for to, ed, sd in zip(toi[:-1], toi[1:], sides)
508        ]

443    def _find_cycles_kinetics(self):
444        """
445        Find the gait cycles using the kinetics algorithm.
446
447        Returns
448        -------
449        None
450
451        Raises
452        ------
453        ValueError
454            If no ground reaction force data is available or no flight phases are found.
455        """
456
457        grf = self.resultant_force
458        if grf is None:
459            raise ValueError("no ground reaction force data available.")
460
461        # get the grf and the latero-lateral COP
462        time = grf.index
463        cop_ml = grf["origin"].copy()[self.lateral_axis].to_numpy().flatten()  # type: ignore
464        vgrf = grf["force"].copy()[self.vertical_axis].to_numpy().flatten()  # type: ignore
465
466        # check if there are flying phases
467        flights = vgrf <= self.ground_reaction_force_threshold
468        if not any(flights):
469            raise ValueError("No flight phases have been found on data.")
470
471        # get the minimum reasonable contact time for each step
472        fsamp = float(1 / np.mean(np.diff(time)))
473        dsamples = int(round(fsamp / 4))
474
475        # get the peaks in the normalized grf, then return toe-offs and foot
476        # strikes
477        grfn = vgrf / np.max(vgrf)
478        toi = []
479        fsi = []
480        pks = find_peaks(grfn, 0.5, dsamples)
481        for pk in pks:
482            to = np.where(grfn[pk:] < self.height_threshold)[0]
483            fs = np.where(grfn[:pk] < self.height_threshold)[0]
484            if len(fs) > 0 and len(to) > 0:
485                toi += [to[0] + pk]
486                if len(toi) > 1:
487                    fsi += [fs[-1]]
488        toi = np.unique(toi)
489        fsi = np.unique(fsi)
490
491        # get the mean latero-lateral position of each contact
492        contacts = [np.arange(i, j + 1) for i, j in zip(fsi, toi[1:])]
493        pos = [np.nanmean(cop_ml[i]) for i in contacts]
494
495        # get the mean value of alternated contacts and set the step sides
496        # accordingly
497        evens = np.mean(pos[0:-1:2])
498        odds = np.mean(pos[1:-1:2])
499        sides = []
500        for i in np.arange(len(pos)):
501            if evens < odds:
502                sides += ["left" if i % 2 == 0 else "right"]
503            else:
504                sides += ["left" if i % 2 != 0 else "right"]
505
506        return [
507            self._get_cycle(float(time[to]), float(time[ed]), sd)
508            for to, ed, sd in zip(toi[:-1], toi[1:], sides)
509        ]
