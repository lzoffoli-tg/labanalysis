 94    def _get_repetitions_start_from_emg_data(
 95        self,
 96        emg: np.ndarray,
 97        time: np.ndarray,
 98    ):
 99        fsamp = int(1 / np.mean(np.diff(time)))
100        batches = sp.continuous_batches(
101            arr=emg > np.max(emg) * 0.1,
102            tolerance=fsamp,
103        )
104        if len(batches) == 0:
105            raise RuntimeError("no repetitions where found on EMG data.")
106        start = int(batches[0][0])
107        while start > 0 and emg[start] > emg[start - 1]:
108            start -= 1
109        return start

111    def _sync(self, force: Signal1D, position: Signal1D, **muscles: EMGSignal):
112        if len(muscles) != 0:
113
114            # get emg data
115            emgs = TimeseriesRecord(
116                **{i: v.copy() for i, v in muscles.items()},
117            )
118            m_time = np.round(np.array(emgs.index) * 1000).astype(int)
119            m_vals = emgs.to_dataframe().sum(axis=1)
120            m_vals = m_vals.values.astype(float).flatten()
121
122            # filter the emg data
123            fsamp_muscles = float(1000 / np.mean(np.diff(m_time)))
124            m_vals -= m_vals.mean()
125            m_vals = sp.butterworth_filt(
126                arr=m_vals,
127                fcut=[20, 450],
128                fsamp=fsamp_muscles,
129                order=4,
130                ftype="bandpass",
131                phase_corrected=True,
132            )
133            # apply teaker-kaiser operator and get linear envelope
134            m_vals = abs(sp.tkeo(m_vals))
135            m_vals = sp.mean_filt(m_vals, int(2 * fsamp_muscles), offset=1)
136
137            # get the first derivative
138            m_der1 = sp.winter_derivative1(m_vals, m_time)
139            m_der1 = np.concatenate([[m_der1[0]], m_der1, [m_der1[-1]]])
140            m_der1 = sp.mean_filt(m_der1, int(2 * fsamp_muscles), offset=1)
141
142            # get force and position data
143            f_time = np.round(np.array(force.index) * 1000).astype(int)
144            f_vals = force.copy().to_numpy()
145            p_vals = position.copy().to_numpy()
146
147            # adjust the force and muscles sampling rate
148            fsamp_biostrength = float(1000 / np.mean(np.diff(f_time)))
149            fsamp = max(fsamp_biostrength, fsamp_muscles)
150            tdiff = 1000 / fsamp
151            m_time_new = np.arange(m_time[0], m_time[-1] + tdiff, tdiff)
152            m_vals = sp.cubicspline_interp(
153                y_old=m_vals,
154                x_old=m_time,
155                x_new=m_time_new,
156            )
157            f_time_new = np.arange(f_time[0], f_time[-1] + tdiff, tdiff)
158            f_vals = sp.cubicspline_interp(
159                y_old=f_vals,
160                x_old=f_time,
161                x_new=f_time_new,
162            )
163            p_vals = sp.cubicspline_interp(
164                y_old=p_vals,
165                x_old=f_time,
166                x_new=f_time_new,
167            )
168
169            # find the start of the repetitions from position
170            pos_start = self._get_repetitions_start_from_biostrength_data(
171                f_vals, p_vals, f_time_new
172            )
173            emg_start = self._get_repetitions_start_from_emg_data(m_vals, m_time_new)
174
175            # set the start to zero on both signals
176            f_time_new = f_time_new - f_time_new[pos_start]
177            m_time = m_time - m_time_new[emg_start]
178            m_time_new = m_time_new - m_time_new[emg_start]
179
180            """
181            # cross-correlate the force and the muscle data corresponding to
182            # the repetitions index
183            xcor, lag = sp.xcorr(
184                sig1=m_vals[m_time_new >= 0],
185                sig2=f_vals[f_time_new >= 0],
186                biased=True,
187                full=True,
188            )
189
190            # smooth the cross-correlation and adjust the emg signal for the lag
191            smoothed = sp.mean_filt(xcor, max(1, int(2 * fsamp)))
192            m_offset = tdiff * lag[np.argmax(smoothed)]
193            m_time_new = m_time_new - m_offset
194            m_time = m_time - m_offset
195            """
196
197            # regenerate the synchronized signals
198            force_sync = Signal1D(
199                f_vals.astype(float),  # type: ignore
200                (f_time_new / 1000).tolist(),  # type: ignore
201                force.unit,
202            )
203            position_sync = Signal1D(
204                p_vals.astype(float),  # type: ignore
205                (f_time_new / 1000).tolist(),  # type: ignore
206                position.unit,
207            )
208            muscles_sync: dict[str, EMGSignal] = {}
209            for muscle, data in muscles.items():
210                m_vals_interp = sp.cubicspline_interp(
211                    data.copy().to_numpy(),
212                    x_old=m_time,
213                    x_new=m_time_new,
214                )
215                new_emgsignal = EMGSignal(
216                    m_vals_interp,  # type: ignore
217                    (m_time_new / 1000).tolist(),  # type: ignore
218                    muscle_name=data.muscle_name,
219                    side=data.side,  # type: ignore
220                    unit=data.unit,
221                )
222                muscles_sync[muscle] = new_emgsignal
223
224            return force_sync, position_sync, muscles_sync
225
226        return force, position, {}

331    def from_txt(
332        cls,
333        filename: str,
334        product: Literal[
335            "LEG PRESS",
336            "LEG PRESS REV",
337            "LEG EXTENSION",
338            "LEG EXTENSION REV",
339            "LEG CURL",
340            "LOW ROW",
341            "ADJUSTABLE PULLEY REV",
342            "CHEST PRESS",
343            "SHOULDER PRESS",
344        ],
345        side: Literal["bilateral", "left", "right"],
346    ):
347        prod = BIOSTRENGTH_PRODUCTS_MAP[product].from_txt_file(filename)
348        load_kgf = prod.load_lever_kgf
349        time_s = prod.time_s
350        pos_m = prod.position_lever_m
351        force = Signal1D(
352            load_kgf * G,
353            time_s,  # type: ignore
354            "N",
355        )
356        position = Signal1D(
357            pos_m,
358            time_s,  # type: ignore
359            "m",
360        )
361        return cls(
362            force=force,
363            position=position,
364            product=prod,
365            side=side,
366        )

368    def to_plotly_figure(self):
369        fig = super().to_plotly_figure()
370        for r, repetition in enumerate(self.repetitions):
371            df = repetition.to_dataframe()
372            for i, (column, values) in enumerate(df.items()):
373                fig.add_trace(
374                    row=i + 1,
375                    col=1,
376                    trace=go.Scatter(
377                        x=df.index.to_list(),
378                        y=values.values.astype(float).flatten().tolist(),
379                        name=f"repetition {r + 1}",
380                        mode="lines",
381                        showlegend=bool(i == 0),
382                        legendgroup=f"repetition {r + 1}",
383                    ),
384                )
385        return fig
