 69    @property
 70    def muscle_symmetry(self):
 71        """
 72        Returns coordination and balance metrics from EMG signals.
 73
 74        Returns
 75        -------
 76        pd.DataFrame
 77            DataFrame with coordination and balance metrics, or empty if not available.
 78        """
 79
 80        # check if a bilateral jump was performed
 81        # (otherwise it makes no sense to test balance)
 82        if self.side != "bilateral":
 83            return pd.DataFrame()
 84
 85        # get the muscle activations
 86        # (if there are no emg data return and empty dataframe)
 87        emgs = self.emgsignals
 88        if emgs.shape[1] == 0:
 89            return pd.DataFrame()
 90
 91        # check the presence of left and right muscles
 92        muscles = {}
 93        for emg in emgs.values():
 94            if isinstance(emg, EMGSignal):
 95                name = emg.muscle_name
 96                side = emg.side
 97                if side not in ["left", "right"]:
 98                    continue
 99                if name not in list(muscles.keys()):
100                    muscles[name] = {}
101
102                # get the area under the curve of the muscle activation
103                muscles[name][side] = emg.to_numpy().flatten()
104
105        # remove those muscles not having both sides
106        muscles = {i: v for i, v in muscles.items() if len(v) == 2}
107
108        # calculate coordination and imbalance between left and right side
109        out = {}
110        for muscle, sides in muscles.items():
111            params = self._get_symmetry(**sides)
112            out.update(**{f"{muscle}_{i}": v for i, v in params.items()})
113
114        return pd.DataFrame(pd.Series(out)).T

116    @property
117    def force_symmetry(self):
118        """
119        Returns coordination and balance metrics from force signals.
120
121        Returns
122        -------
123        pd.DataFrame
124            DataFrame with force coordination and balance metrics, or empty if not available.
125        """
126
127        # get the forces from each foot and hand
128        left_foot = self.get("left_foot_ground_reaction_force")
129        right_foot = self.get("right_foot_ground_reaction_force")
130        if (
131            left_foot is None
132            or right_foot is None
133            or not isinstance(left_foot, ForcePlatform)
134            or not isinstance(right_foot, ForcePlatform)
135        ):
136            return pd.DataFrame()
137        left_vt = left_foot.copy().force[self.vertical_axis].to_numpy().flatten()  # type: ignore
138        right_vt = right_foot.copy().force[self.vertical_axis].to_numpy().flatten()  # type: ignore
139
140        # get the pairs to be tested
141        pairs = {
142            "lower_limbs": {"left_foot": left_vt, "right_foot": right_vt},
143        }
144
145        # calculate balance and coordination
146        out = []
147        unit = self.resultant_force
148        if unit is None:
149            return pd.DataFrame()
150        unit = unit.force.unit
151        for region, pair in pairs.items():
152            left, right = list(pair.values())
153            fit = self._get_symmetry(left, right)
154            line = {f"force_{i}": float(v) for i, v in fit.items()}  # type: ignore
155            line = pd.DataFrame(pd.Series(line)).T
156            line.insert(0, "region", region)
157            out += [line]
158
159        return pd.concat(out, ignore_index=True)

161    @property
162    def area_of_stability_sqm(self):
163        cop = self.resultant_force
164        if cop is None:
165            return np.nan
166        cop = cop.origin.to_dataframe().dropna()
167        horizontal_axes = [self.anteroposterior_axis, self.lateral_axis]
168        ap, ml = cop[horizontal_axes].values.astype(float).T
169        ellipse = Ellipse().fit(ap.astype(float), ml.astype(float))
170        return ellipse.area

172    def _get_symmetry(self, left: np.ndarray, right: np.ndarray):
173        line = {"left_%": np.mean(left), "right_%": np.mean(right)}
174        line = pd.DataFrame(pd.Series(line)).T
175        norm = line.sum(axis=1).values.astype(float)
176        line.loc[line.index, line.columns] = line.values.astype(float) / norm * 100
177        return line

227            cop = cop.copy()["origin"].to_dataframe().dropna()

229            out["analytics"]["centre_of_pressure"] = cop
230        return out

A dictionary-like container for Timeseries, TimeseriesRecord, and ForcePlatform objects, supporting type filtering and DataFrame conversion.

Parameters
  • vertical_axis (str, optional): The label for the vertical axis (default "Y").
  • anteroposterior_axis (str, optional): The label for the anteroposterior axis (default "Z").
  • strip (bool, optional): If True, remove leading/trailing rows or columns that are all NaN from all contained objects (default True).
  • reset_time (bool, optional): If True, reset the time index to start at zero for all contained objects (default True).
  • **signals (dict): Key-value pairs of Timeseries subclasses, TimeseriesRecord, or ForcePlatform to include in the record.
Attributes
  • _vertical_axis (str): The vertical axis label.

 84            return pd.DataFrame()

 87        # (if there are no emg data return and empty dataframe)

 97                side = emg.side
 98                if side not in ["left", "right"]:
 99                    continue

102
