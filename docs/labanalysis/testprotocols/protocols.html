661        """
662        Returns the normative values loaded from the CSV file.
663
664        Returns
665        -------
666        pandas.DataFrame
667            The normative values loaded from the CSV file.
668        """
669        if self.normative_data_path == "":
670            return pd.DataFrame()
671
672        data = pd.read_csv(self.normative_data_path)
673        for row, line in data.iterrows():
674            row = int(row)  # type: ignore
675            ranges = np.arange(100)
676            percs = normal_distribution.ppf(
677                ranges / 100,
678                loc=line["mean"],
679                scale=line["std"],
680            )
681            labels = [f"{i:02}" for i in ranges]
682            data.loc[row, labels] = percs
683        return data
